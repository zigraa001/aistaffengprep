# Recursive DFS
def dfs_recursive(graph, node, visited=None):
    if visited is None:
        visited = set()

    visited.add(node)
    print(node)  # Process node

    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs_recursive(graph, neighbor, visited)

# Iterative DFS (with stack)
def dfs_iterative(graph, start):
    visited = set()
    stack = [start]

    while stack:
        node = stack.pop()
        if node in visited:
            continue

        visited.add(node)
        print(node)  # Process node

        for neighbor in graph[node]:
            if neighbor not in visited:
                stack.append(neighbor)

# Number of Islands (classic DFS problem)
def num_islands(grid):
    if not grid:
        return 0

    rows, cols = len(grid), len(grid[0])
    count = 0

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != '1':
            return
        grid[r][c] = '0'  # Mark visited
        dfs(r+1, c)
        dfs(r-1, c)
        dfs(r, c+1)
        dfs(r, c-1)

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1':
                dfs(r, c)
                count += 1

    return count

BFS for Shortest Path ▼ 
from collections import deque

def bfs(graph, start):
    visited = {start}
    queue = deque([start])

    while queue:
        node = queue.popleft()
        print(node)  # Process node

        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

# Shortest path in unweighted graph
def shortest_path(graph, start, end):
    if start == end:
        return [start]

    visited = {start}
    queue = deque([(start, [start])])

    while queue:
        node, path = queue.popleft()

        for neighbor in graph[node]:
            if neighbor == end:
                return path + [neighbor]
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append((neighbor, path + [neighbor]))

    return []  # No path found

When to Use

  • DFS: Path finding, cycle detection, topological sort, exhaustive search
  • BFS: Shortest path (unweighted), level-order, nearest neighbor

Topological Sort

Kahn's Algorithm (BFS-based) ▼ 
from collections import defaultdict, deque

def topological_sort(num_nodes, edges):
    """
    Returns topological ordering or empty list if cycle exists.
    Used for: Course scheduling, build systems, dependency resolution
    """
    # Build graph and count in-degrees
    graph = defaultdict(list)
    in_degree = [0] * num_nodes

    for u, v in edges:  # u -> v (u must come before v)
        graph[u].append(v)
        in_degree[v] += 1

    # Start with nodes that have no prerequisites
    queue = deque([i for i in range(num_nodes) if in_degree[i] == 0])
    result = []

    while queue:
        node = queue.popleft()
        result.append(node)

        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # If result doesn't contain all nodes, there's a cycle
    return result if len(result) == num_nodes else []

# Course Schedule: Can finish all courses?
def can_finish(num_courses, prerequisites):
    return len(topological_sort(num_courses, prerequisites)) == num_courses

Union-Find (Disjoint Set)

Implementation with Optimizations ▼ 
class UnionFind:
    """
    Union-Find with path compression and union by rank.
    Time: O(α(n)) ≈ O(1) per operation
    """
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.components = n

    def find(self, x):
        """Find root with path compression."""
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):
        """Union by rank. Returns True if merged, False if already connected."""
        px, py = self.find(x), self.find(y)

        if px == py:
            return False

        # Attach smaller tree to larger tree
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px

        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1

        self.components -= 1
        return True

    def connected(self, x, y):
        return self.find(x) == self.find(y)

# Number of Connected Components
def count_components(n, edges):
    uf = UnionFind(n)
    for u, v in edges:
        uf.union(u, v)
    return uf.components

Real-World Scale Examples

LinkedIn: Graph Algorithms at 1B+ Users ▼

LinkedIn's social graph has over 1 billion members and 67+ million companies. Their graph infrastructure processes trillions of edges for features like "People You May Know" and connection path finding.

The "2nd Degree Connection" Problem

  • Scale: Average user has 500 connections, each has 500 more
  • Challenge: Finding mutual connections = BFS to depth 2
  • Solution: Pre-compute and cache 2nd-degree neighborhoods
# LinkedIn-style "People You May Know" simplified
from collections import defaultdict

class SocialGraph:
    """
    LinkedIn processes 300M+ API calls/day for connection queries.
    Key insight: pre-compute what you query frequently.
    """
    def __init__(self):
        self.connections = defaultdict(set)  # user -> set of connections
        self.second_degree_cache = {}  # Pre-computed for active users

    def add_connection(self, user_a, user_b):
        self.connections[user_a].add(user_b)
        self.connections[user_b].add(user_a)
        # Invalidate cache for both users
        self.second_degree_cache.pop(user_a, None)
        self.second_degree_cache.pop(user_b, None)

    def get_second_degree(self, user):
        """
        BFS to depth 2. LinkedIn caches this for active users.
        Without caching: O(connections * avg_connections) per request.
        """
        if user in self.second_degree_cache:
            return self.second_degree_cache[user]

        first_degree = self.connections[user]
        second_degree = set()

        for friend in first_degree:
            for friend_of_friend in self.connections[friend]:
                if friend_of_friend != user and friend_of_friend not in first_degree:
                    second_degree.add(friend_of_friend)

        self.second_degree_cache[user] = second_degree
        return second_degree

    def people_you_may_know(self, user, limit=10):
        """
        Rank 2nd-degree connections by mutual connection count.
        LinkedIn uses ML models on top of this base signal.
        """
        second_degree = self.get_second_degree(user)
        first_degree = self.connections[user]

        # Count mutual connections
        scores = {}
        for candidate in second_degree:
            mutual = len(self.connections[candidate] & first_degree)
            scores[candidate] = mutual

        # Return top candidates by mutual connections
        ranked = sorted(scores.items(), key=lambda x: -x[1])
        return [user for user, score in ranked[:limit]]
Google Maps: Shortest Path at Global Scale ▼

Google Maps serves over 1 billion users, processing 1 billion kilometers of directions daily. The road network graph has billions of nodes (intersections) and edges (road segments).

Why Not Just Dijkstra?

  • Problem: Dijkstra explores in all directions until finding destination
  • At scale: NYC to LA would explore entire US road network
  • Solution: Hierarchical graphs + precomputation
# Simplified Contraction Hierarchies (used by Google Maps, OSRM)
import heapq

class ContractionHierarchies:
    """
    Preprocess graph to enable bidirectional search on "important" nodes only.
    Preprocessing: expensive (minutes to hours for large road networks)
    Query: sub-millisecond in practice, vs O((V+E) log V) for Dijkstra
    """
    def __init__(self, graph):
        self.graph = graph
        self.importance = {}  # Node importance ranking
        self.shortcuts = defaultdict(list)  # Added shortcut edges

    def preprocess(self):
        """
        Contract nodes in order of importance.
        Add shortcuts to preserve shortest paths.
        Google does this offline - takes hours for continent-scale.
        """
        # Simplified: importance = degree (real algorithm more complex)
        nodes = sorted(self.graph.keys(),
                      key=lambda n: len(self.graph[n]))

        for i, node in enumerate(nodes):
            self.importance[node] = i
            self._contract_node(node)

    def _contract_node(self, node):
        """Add shortcuts between neighbors if node is on shortest path."""
        neighbors = list(self.graph[node])
        for u in neighbors:
            for v in neighbors:
                if u != v:
                    # Check if shortest u->v goes through node
                    uv_through_node = (
                        self._edge_weight(u, node) +
                        self._edge_weight(node, v)

                    direct_uv = self._edge_weight(u, v)

                        # Add shortcut

Facebook: Graph Partitioning for Social Feed ▼

Facebook's social graph has 3 billion users and trillions of edges. Storing this on a single machine is impossible. Graph partitioning determines which users are stored together.

The Challenge

  • Goal: Minimize cross-machine queries when loading a user's feed
  • Constraint: Balanced partition sizes for even load distribution
  • This is NP-hard: Facebook uses approximation algorithms
# Simplified graph partitioning (real version uses METIS, Pregel)
class GraphPartitioner:
    """
    Assign users to servers to minimize cross-server edges.
    Facebook re-partitions periodically as friendships change.
    """
    def __init__(self, graph, num_partitions):
        self.graph = graph
        self.num_partitions = num_partitions
        self.assignment = {}  # node -> partition

    def partition_balanced(self):
        """
        Greedy balanced partitioning.
        Assign each node to partition with most existing friends.
        """
        partition_sizes = [0] * self.num_partitions
        max_size = len(self.graph) // self.num_partitions + 1

        for node in self.graph:
            # Count friends in each partition
            friend_counts = [0] * self.num_partitions
            for friend in self.graph[node]:
                if friend in self.assignment:
                    friend_counts[self.assignment[friend]] += 1

            # Assign to partition with most friends (if not full)
            for p in sorted(range(self.num_partitions),
                          key=lambda x: -friend_counts[x]):
                if partition_sizes[p] < max_size:
                    self.assignment[node] = p
                    partition_sizes[p] += 1
                    break

    def edge_cut_ratio(self):
        """Measure partitioning quality: % of edges crossing partitions."""
        cross_edges = 0
        total_edges = 0

        for node in self.graph:
            for neighbor in self.graph[node]:
                total_edges += 1
                if self.assignment[node] != self.assignment[neighbor]:
                    cross_edges += 1

        return cross_edges / total_edges if total_edges else 0

# Facebook achieves ~10% edge cut - 90% of friend data is local!

Shortest Path Algorithms

Shortest path is one of the most frequently tested graph topics at staff level. You must know when each algorithm applies, why it works, and where it breaks.

Dijkstra's Algorithm ▼

Core idea: Greedily expand the nearest unvisited node. Dijkstra works because once a node is finalized (popped from the min-heap), no shorter path to it can exist -- but only when all edge weights are non-negative.

Why Dijkstra Fails with Negative Edges

Dijkstra assumes that once a node is "settled" (removed from the priority queue), its shortest distance is final. A negative-weight edge discovered later could provide a shorter path to an already-settled node, violating this invariant. Consider: A->B (weight 1), A->C (weight 5), C->B (weight -10). Dijkstra settles B at distance 1, but the real shortest is A->C->B = -5.

Complexity

  • With binary heap: O((V + E) log V) -- the standard interview version
  • With Fibonacci heap: O(V log V + E) -- theoretical improvement, rarely used in practice
  • Space: O(V) for distances + O(V) for the heap
import heapq
from collections import defaultdict

def dijkstra(graph, start):
    """
    Dijkstra's shortest path from start to all other nodes.
    graph: dict of {node: [(neighbor, weight), ...]}
    Returns: dict of {node: shortest_distance}

    Time:  O((V + E) log V) with binary min-heap
    Space: O(V)
    """
    dist = {start: 0}
    heap = [(0, start)]  # (distance, node)

    while heap:
        d, u = heapq.heappop(heap)

        # Key optimization: skip stale entries
        # When we push a better distance for a node, the old entry
        # remains in the heap. This check discards it.
        if d > dist.get(u, float('inf')):
            continue

        for v, weight in graph[u]:
            new_dist = d + weight
            if new_dist < dist.get(v, float('inf')):
                dist[v] = new_dist
                heapq.heappush(heap, (new_dist, v))

    return dist


def dijkstra_with_path(graph, start, end):
    """
    Returns (shortest_distance, path) from start to end.
    Useful for interview problems that ask for the actual path.
    """
    dist = {start: 0}
    prev = {start: None}
    heap = [(0, start)]

    while heap:
        d, u = heapq.heappop(heap)

        if u == end:
            # Reconstruct path
            path = []
            while u is not None:
                path.append(u)
                u = prev[u]
            return d, path[::-1]

        if d > dist.get(u, float('inf')):
            continue

        for v, weight in graph[u]:
            new_dist = d + weight
            if new_dist < dist.get(v, float('inf')):
                dist[v] = new_dist
                prev[v] = u
                heapq.heappush(heap, (new_dist, v))

    return float('inf'), []  # No path


# ---------- Example usage ----------
# LeetCode 743: Network Delay Time
def network_delay_time(times, n, k):
    """
    Given n nodes and directed weighted edges 'times',
    return time for signal from node k to reach ALL nodes.
    If impossible, return -1.
    """
    graph = defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))

    dist = dijkstra(graph, k)

    if len(dist) != n:
        return -1  # Some nodes unreachable

    return max(dist.values())

Interview tip: Always mention the "stale entry" optimization (checking if d > dist[u]) when discussing Dijkstra. It shows you understand why duplicates appear in the heap and how to handle them without a decrease-key operation.

Bellman-Ford Algorithm ▼

Core idea: Relax all edges V-1 times. After V-1 iterations, shortest paths using at most V-1 edges are found. A V-th pass that still relaxes an edge proves a negative cycle exists.

Why V-1 Iterations?

In a graph with V nodes, the longest possible shortest path (without negative cycles) has at most V-1 edges. Each iteration of Bellman-Ford extends shortest paths by one more edge. So after V-1 iterations, we've found all shortest paths.

Complexity

  • Time: O(V * E) -- slower than Dijkstra but handles negative weights
  • Space: O(V)
def bellman_ford(n, edges, start):
    """
    Single-source shortest path with negative edge support.
    edges: list of (u, v, weight)
    Returns: (dist dict, has_negative_cycle bool)

    Time:  O(V * E)
    Space: O(V)
    """
    dist = [float('inf')] * n
    dist[start] = 0

    # Relax all edges V-1 times
    for i in range(n - 1):
        updated = False
        for u, v, w in edges:
            if dist[u] != float('inf') and dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                updated = True
        # Early termination: no updates means we're done
        if not updated:
            break

    # V-th pass: check for negative cycles
    for u, v, w in edges:
        if dist[u] != float('inf') and dist[u] + w < dist[v]:
            return dist, True  # Negative cycle detected

    return dist, False


# ---------- Classic application: Cheapest Flights Within K Stops ----------
# LeetCode 787
def find_cheapest_price(n, flights, src, dst, k):
    """
    Modified Bellman-Ford: only k+1 relaxation rounds (k stops = k+1 edges).
    This is the cleanest way to solve "shortest path with at most K edges."
    """
    dist = [float('inf')] * n
    dist[src] = 0

    for i in range(k + 1):
        # CRITICAL: use a copy so we don't use updates from this round
        temp = dist[:]
        for u, v, w in flights:
            if dist[u] != float('inf') and dist[u] + w < temp[v]:
                temp[v] = dist[u] + w
        dist = temp

    return dist[dst] if dist[dst] != float('inf') else -1

Interview tip: The "Cheapest Flights Within K Stops" variant is extremely popular. The key insight is using a copy of the distance array each round so you only consider paths with exactly one more edge, not paths updated within the same iteration.

Floyd-Warshall Algorithm ▼

Core idea: Dynamic programming over intermediate nodes. For each pair (i, j), check if routing through node k gives a shorter path. After considering all k, we have all-pairs shortest paths.

When to Use Floyd-Warshall

  • You need shortest paths between all pairs of nodes (not just from one source)
  • The graph is dense (E is close to V^2)
  • V is small (typically V <= 400 for competitive programming, due to O(V^3))
  • You need to detect negative cycles (check if dist[i][i] < 0 for any i)

Complexity

  • Time: O(V^3)
  • Space: O(V^2)
def floyd_warshall(n, edges):
    """
    All-pairs shortest paths.
    edges: list of (u, v, weight)
    Returns: n x n distance matrix

    Time:  O(V^3)
    Space: O(V^2)
    """
    INF = float('inf')
    dist = [[INF] * n for _ in range(n)]

    # Distance to self is 0
    for i in range(n):
        dist[i][i] = 0

    # Initialize with direct edges
    for u, v, w in edges:
        dist[u][v] = min(dist[u][v], w)  # Handle multiple edges

    # DP: try each node k as intermediate
    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] + dist[k][j] < dist[i][j]:
                    dist[i][j] = dist[i][k] + dist[k][j]

    # Detect negative cycles: dist[i][i] < 0
    for i in range(n):
        if dist[i][i] < 0:
            return dist, True  # Negative cycle exists

    return dist, False


# ---------- Application: Find the City With Smallest Number of Neighbors ----------
# LeetCode 1334
def find_the_city(n, edges, distance_threshold):
    """
    Find city with fewest reachable cities within distance_threshold.
    Floyd-Warshall is perfect here: small n, need all-pairs distances.
    """
    dist, _ = floyd_warshall(n, edges)

    # Also add reverse edges for undirected graph
    INF = float('inf')
    d = [[INF] * n for _ in range(n)]
    for i in range(n):
        d[i][i] = 0
    for u, v, w in edges:
        d[u][v] = min(d[u][v], w)
        d[v][u] = min(d[v][u], w)
    for k in range(n):
        for i in range(n):
            for j in range(n):
                d[i][j] = min(d[i][j], d[i][k] + d[k][j])

    result = -1
    min_reachable = n + 1
    for i in range(n):
        reachable = sum(1 for j in range(n) if i != j and d[i][j] <= distance_threshold)
        if reachable <= min_reachable:
            min_reachable = reachable
            result = i

    return result

  • All-pairs and V is small (<= 400)? Use Floyd-Warshall.

Minimum Spanning Tree (MST)

An MST connects all nodes with the minimum total edge weight, using exactly V-1 edges. Two classic algorithms exist, and interviewers expect you to know both and when each is preferable.

Kruskal's Algorithm (Union-Find Based) ▼

Core idea: Sort all edges by weight. Greedily add the smallest edge that does not form a cycle (checked via Union-Find). Stop after V-1 edges.

Why It Works (Cut Property)

The cut property states: for any cut of the graph, the minimum weight edge crossing the cut must be in the MST. Since Kruskal processes edges in order, each edge it adds is the minimum crossing some cut between two components.

Complexity

  • Time: O(E log E) for sorting, then O(E * alpha(V)) for union-find operations. Total: O(E log E)
  • Space: O(V) for the Union-Find structure
  • Best when: Graph is sparse (E is much less than V^2), or edges are already sorted
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return False
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        return True


def kruskal(n, edges):
    """
    Kruskal's MST algorithm.
    edges: list of (weight, u, v)
    Returns: (mst_weight, mst_edges)

    Time:  O(E log E)
    Space: O(V)
    """
    edges.sort()  # Sort by weight
    uf = UnionFind(n)
    mst_weight = 0
    mst_edges = []

    for weight, u, v in edges:
        if uf.union(u, v):
            mst_weight += weight
            mst_edges.append((u, v, weight))
            if len(mst_edges) == n - 1:
                break  # MST complete

    # If len(mst_edges) < n-1, graph is not connected
    return mst_weight, mst_edges


# ---------- LeetCode 1135: Connecting Cities With Minimum Cost ----------
def minimum_cost(n, connections):
    """
    connections: [[city1, city2, cost], ...]
    Return min cost to connect all cities, or -1 if impossible.
    """
    edges = [(cost, c1 - 1, c2 - 1) for c1, c2, cost in connections]
    total, mst = kruskal(n, edges)
    return total if len(mst) == n - 1 else -1

def min_cost_connect_points(points):
    """
    Points on 2D plane, cost = Manhattan distance.
    This is an MST problem: connect all points with minimum total distance.
    """
    n = len(points)
    edges = []
    for i in range(n):
        for j in range(i + 1, n):
            dist = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])
            edges.append((dist, i, j))

    total, _ = kruskal(n, edges)
    return total
Prim's Algorithm ▼

Core idea: Start from any node. Greedily add the cheapest edge connecting the current MST to a node not yet in the MST. Like Dijkstra but tracking edge weights rather than total path distance.

Complexity

  • With binary heap: O((V + E) log V)
  • With Fibonacci heap: O(E + V log V) -- better for dense graphs
  • Best when: Graph is dense (E close to V^2), or you already have an adjacency list
import heapq

def prim(n, adj):
    """
    Prim's MST algorithm.
    adj: dict of {node: [(neighbor, weight), ...]}
    Returns: total MST weight

    Time:  O((V + E) log V)
    Space: O(V)
    """
    in_mst = [False] * n
    # (weight, node) -- start from node 0
    heap = [(0, 0)]
    mst_weight = 0
    edges_added = 0

    while heap and edges_added < n:
        weight, u = heapq.heappop(heap)

        if in_mst[u]:
            continue

        in_mst[u] = True
        mst_weight += weight
        edges_added += 1

        for v, w in adj[u]:
            if not in_mst[v]:
                heapq.heappush(heap, (w, v))

    return mst_weight if edges_added == n else -1  # -1 if disconnected


# ---------- Prim's for Min Cost to Connect All Points ----------
def min_cost_connect_points_prim(points):
    """
    For dense graphs (like complete graphs), Prim's can be more efficient
    because we avoid sorting O(V^2) edges explicitly.
    """
    n = len(points)
    adj = {i: [] for i in range(n)}
    for i in range(n):
        for j in range(i + 1, n):
            dist = abs(points[i][0] - points[j][0]) + abs(points[i][1] - points[j][1])
            adj[i].append((j, dist))
            adj[j].append((i, dist))

    return prim(n, adj)

Graph Coloring & Bipartite Graphs

Bipartite Graph Check (2-Coloring) ▼

Definition: A graph is bipartite if you can color every node with one of two colors such that no two adjacent nodes share the same color. Equivalently, it has no odd-length cycles.

Why It Matters

  • Matching problems: Bipartite matching is a fundamental problem (job assignment, stable marriage)
  • Scheduling: Can a set of constraints be split into two groups?
  • Interview frequency: LeetCode #785 "Is Graph Bipartite?" appears frequently at FAANG
from collections import deque

def is_bipartite(graph):
    """
    LeetCode 785: Is Graph Bipartite?
    graph[i] = list of neighbors of node i.

    BFS approach: try to 2-color the graph.
    If any neighbor has the same color, it's not bipartite.

    Time:  O(V + E)
    Space: O(V)
    """
    n = len(graph)
    color = [-1] * n  # -1 = uncolored, 0 = color A, 1 = color B

    for start in range(n):
        if color[start] != -1:
            continue

        # BFS from this unvisited node
        queue = deque([start])
        color[start] = 0

        while queue:
            node = queue.popleft()
            for neighbor in graph[node]:
                if color[neighbor] == -1:
                    color[neighbor] = 1 - color[node]  # Opposite color
                    queue.append(neighbor)
                elif color[neighbor] == color[node]:
                    return False  # Same color = odd cycle = not bipartite

    return True


# DFS version (equally valid, sometimes more natural)
def is_bipartite_dfs(graph):
    n = len(graph)
    color = [-1] * n

    def dfs(node, c):
        color[node] = c
        for neighbor in graph[node]:
            if color[neighbor] == -1:
                if not dfs(neighbor, 1 - c):
                    return False
            elif color[neighbor] == c:
                return False
        return True

    return all(
        color[i] != -1 or dfs(i, 0)
        for i in range(n)
    )

Graph Coloring (General)

General graph coloring (using k colors) is NP-complete for k >= 3. In interviews, you will almost always see the 2-coloring (bipartite) case. If asked about k-coloring, discuss:

  • Greedy coloring: Assign the smallest available color to each node. Uses at most D+1 colors (D = max degree). Not optimal but fast.
  • Backtracking: Try all color assignments. Exponential but correct for small graphs.
  • Real-world use: Register allocation in compilers, map coloring, scheduling exams so no student has two at the same time.
def greedy_coloring(graph):
    """
    Greedy graph coloring. Assigns smallest available color to each node.
    Uses at most max_degree + 1 colors.
    Time: O(V + E)
    """
    n = len(graph)
    result = [-1] * n

    for node in range(n):
        # Find colors used by neighbors
        used = set()
        for neighbor in graph[node]:
            if result[neighbor] != -1:
                used.add(result[neighbor])

        # Assign smallest available color
        color = 0
        while color in used:
            color += 1
        result[node] = color

    return result  # result[i] = color assigned to node i

Number of Islands (#200) -- Three Approaches Compared ▼

This is the single most asked graph problem. Interviewers use it as a warmup and then layer on follow-ups. Know all three approaches and their trade-offs.

# Approach 1: DFS (most intuitive, shown above in DFS section)
# Time: O(M*N), Space: O(M*N) worst case for recursion stack
# Modifies input grid -- discuss this trade-off!

# Approach 2: BFS (iterative, avoids stack overflow on large grids)
from collections import deque

def num_islands_bfs(grid):
    """
    BFS avoids recursion limit issues on very large grids.
    Python's default recursion limit is 1000, and a grid of 300x300
    could have a DFS depth of 90,000.
    """
    if not grid:
        return 0

    rows, cols = len(grid), len(grid[0])
    count = 0

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1':
                count += 1
                grid[r][c] = '0'
                queue = deque([(r, c)])
                while queue:
                    cr, cc = queue.popleft()
                    for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
                        nr, nc = cr + dr, cc + dc
                        if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == '1':
                            grid[nr][nc] = '0'
                            queue.append((nr, nc))
    return count


# Approach 3: Union-Find (best for follow-ups like "stream of islands")
class UnionFindGrid:
    def __init__(self, grid):
        m, n = len(grid), len(grid[0])
        self.parent = {}
        self.rank = {}
        self.count = 0

        for r in range(m):
            for c in range(n):
                if grid[r][c] == '1':
                    self.parent[(r, c)] = (r, c)
                    self.rank[(r, c)] = 0
                    self.count += 1

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py:
            return
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        self.count -= 1


def num_islands_uf(grid):
    if not grid:
        return 0

    m, n = len(grid), len(grid[0])
    uf = UnionFindGrid(grid)

    for r in range(m):
        for c in range(n):
            if grid[r][c] == '1':
                for dr, dc in [(1, 0), (0, 1)]:  # Only right and down
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] == '1':
                        uf.union((r, c), (nr, nc))

    return uf.count
Approach Time Space Pros Cons
DFS O(M*N) O(M*N) Simplest to write Stack overflow on large grids
BFS O(M*N) O(min(M,N)) No stack overflow; better space Slightly more code
Union-Find O(M*N * alpha) O(M*N) Handles streaming updates More complex implementation

Follow-up interviewers love: "What if islands arrive as a stream?" (LeetCode #305). Union-Find is the only approach that handles this efficiently, since DFS/BFS require re-scanning the entire grid.

Course Schedule I & II -- Topological Sort Application ▼

These problems test your understanding of directed graphs and topological ordering. Course Schedule I asks "is it possible?" and Course Schedule II asks "in what order?"

from collections import defaultdict, deque

# LeetCode 207: Course Schedule I
def can_finish(num_courses, prerequisites):
    """
    Can you finish all courses? Equivalent to: does the dependency graph have a cycle?
    If topological sort includes all nodes, no cycle exists.
    """
    graph = defaultdict(list)
    in_degree = [0] * num_courses

    for course, prereq in prerequisites:
        graph[prereq].append(course)
        in_degree[course] += 1

    queue = deque(i for i in range(num_courses) if in_degree[i] == 0)
    completed = 0

    while queue:
        node = queue.popleft()
        completed += 1
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    return completed == num_courses


# LeetCode 210: Course Schedule II
def find_order(num_courses, prerequisites):
    """
    Return a valid ordering, or [] if impossible.
    This IS the topological sort result.
    """
    graph = defaultdict(list)
    in_degree = [0] * num_courses

    for course, prereq in prerequisites:
        graph[prereq].append(course)
        in_degree[course] += 1

    queue = deque(i for i in range(num_courses) if in_degree[i] == 0)
    order = []

    while queue:
        node = queue.popleft()
        order.append(node)
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    return order if len(order) == num_courses else []


# DFS-based topological sort (alternative approach)
def find_order_dfs(num_courses, prerequisites):
    """
    DFS post-order gives reverse topological order.
    Use 3-color marking to detect cycles:
      0 = unvisited, 1 = in current path (gray), 2 = completed (black)
    """
    graph = defaultdict(list)
    for course, prereq in prerequisites:
        graph[prereq].append(course)

    state = [0] * num_courses  # 0=white, 1=gray, 2=black
    result = []

    def dfs(node):
        if state[node] == 1:  # Back edge = cycle
            return False
        if state[node] == 2:  # Already processed
            return True

        state[node] = 1  # Mark as in-progress
        for neighbor in graph[node]:
            if not dfs(neighbor):
                return False

        state[node] = 2  # Mark as completed
        result.append(node)  # Post-order
        return True

    for i in range(num_courses):
        if state[i] == 0:
            if not dfs(i):
                return []

    return result[::-1]  # Reverse post-order = topological order

Interview tip: Know BOTH approaches (Kahn's BFS and DFS post-order). Kahn's is simpler to implement correctly; DFS is more elegant and the 3-color scheme generalizes to many problems. If asked "what if there are multiple valid orderings?", Kahn's makes it easy to use a priority queue to get lexicographically smallest order.

Clone Graph (#133) ▼

This problem tests deep copy of a graph. The key insight is using a hash map from original node to cloned node to handle cycles and shared references.

class Node:
    def __init__(self, val=0, neighbors=None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []


def clone_graph(node):
    """
    LeetCode 133: Clone Graph
    Deep copy an undirected graph. Handle cycles with a visited map.

    Time:  O(V + E)
    Space: O(V)
    """
    if not node:
        return None

    cloned = {}  # original node -> cloned node

    def dfs(original):
        if original in cloned:
            return cloned[original]

        copy = Node(original.val)
        cloned[original] = copy  # Map BEFORE recursing (handles cycles)

        for neighbor in original.neighbors:
            copy.neighbors.append(dfs(neighbor))

        return copy

    return dfs(node)


# BFS version
from collections import deque

def clone_graph_bfs(node):
    if not node:
        return None

    cloned = {node: Node(node.val)}
    queue = deque([node])

    while queue:
        original = queue.popleft()
        for neighbor in original.neighbors:
            if neighbor not in cloned:
                cloned[neighbor] = Node(neighbor.val)
                queue.append(neighbor)
            cloned[original].neighbors.append(cloned[neighbor])

    return cloned[node]

Pattern: "Clone X" problems (graph, linked list with random pointer, tree) all use the same approach: a hash map from original to copy, creating copies on first visit and looking up on subsequent visits. The map is what prevents infinite loops on cycles.

Word Ladder (#127) -- BFS Shortest Path ▼

An implicit graph problem: each word is a node, and edges connect words that differ by one character. Find the shortest transformation sequence. This is a classic BFS shortest path problem disguised as a string problem.

from collections import deque, defaultdict

def ladder_length(begin_word, end_word, word_list):
    """
    LeetCode 127: Word Ladder
    Find shortest transformation sequence from begin to end,
    changing one letter at a time, using only words in word_list.

    Key insight: Build an implicit graph using wildcard patterns.
    "hot" matches pattern "*ot", "h*t", "ho*"

    Time:  O(M^2 * N) where M = word length, N = word list size
    Space: O(M^2 * N) for the pattern map
    """
    if end_word not in word_list:
        return 0

    word_set = set(word_list)
    word_len = len(begin_word)

    # Build adjacency via wildcard patterns
    # "hot" -> {"*ot": ["hot"], "h*t": ["hot"], "ho*": ["hot"]}
    pattern_map = defaultdict(list)
    for word in word_set:
        for i in range(word_len):
            pattern = word[:i] + '*' + word[i+1:]
            pattern_map[pattern].append(word)

    # BFS
    queue = deque([(begin_word, 1)])
    visited = {begin_word}

    while queue:
        word, steps = queue.popleft()

        for i in range(word_len):
            pattern = word[:i] + '*' + word[i+1:]

            for neighbor in pattern_map[pattern]:
                if neighbor == end_word:
                    return steps + 1
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, steps + 1))

            # Clear pattern to avoid revisiting
            pattern_map[pattern] = []

    return 0  # No transformation possible


# ---------- Optimization: Bidirectional BFS ----------
def ladder_length_bidir(begin_word, end_word, word_list):
    """
    Bidirectional BFS: search from both ends simultaneously.
    Meets in the middle, reducing search space from O(b^d) to O(b^(d/2)).

    This is the optimization interviewers love to see discussed.
    """
    word_set = set(word_list)
    if end_word not in word_set:
        return 0

    front = {begin_word}
    back = {end_word}
    visited = set()
    word_len = len(begin_word)
    steps = 1

    while front and back:
        # Always expand the smaller frontier
        if len(front) > len(back):
            front, back = back, front

        next_front = set()
        for word in front:
            for i in range(word_len):
                for c in 'abcdefghijklmnopqrstuvwxyz':
                    new_word = word[:i] + c + word[i+1:]

                    if new_word in back:
                        return steps + 1

                    if new_word in word_set and new_word not in visited:
                        visited.add(new_word)
                        next_front.add(new_word)

        front = next_front
        steps += 1

    return 0

Interview tip: Always mention bidirectional BFS for Word Ladder. Even if you don't implement it fully, discussing the optimization from O(b^d) to O(b^(d/2)) shows strong algorithm intuition. This pattern applies to any BFS problem where both start and end are known.

Alien Dictionary (#269) -- Topological Sort on Characters ▼

Given a sorted list of words in an alien language, determine the ordering of characters. This is a topological sort problem where you derive the graph from the sorted word order.

from collections import defaultdict, deque

def alien_order(words):
    """
    LeetCode 269: Alien Dictionary

    Key insight: Compare adjacent words to find character ordering.
    If words[i] = "abc" and words[i+1] = "abd", then c < d.
    Only the FIRST differing character gives us information.

    Time:  O(C) where C = total characters across all words
    Space: O(1) since alphabet size is bounded (26 letters max)
    """
    # Step 1: Build graph from adjacent word comparisons
    graph = defaultdict(set)
    in_degree = {c: 0 for word in words for c in word}

    for i in range(len(words) - 1):
        w1, w2 = words[i], words[i + 1]
        min_len = min(len(w1), len(w2))

        # Edge case: "abc" before "ab" is INVALID
        if len(w1) > len(w2) and w1[:min_len] == w2[:min_len]:
            return ""

        for j in range(min_len):
            if w1[j] != w2[j]:
                # w1[j] comes before w2[j] in alien alphabet
                if w2[j] not in graph[w1[j]]:
                    graph[w1[j]].add(w2[j])
                    in_degree[w2[j]] += 1
                break  # Only first difference matters!

    # Step 2: Topological sort (Kahn's)
    queue = deque(c for c in in_degree if in_degree[c] == 0)
    result = []

    while queue:
        c = queue.popleft()
        result.append(c)
        for neighbor in graph[c]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor)

    # If not all characters included, there's a cycle (invalid input)
    if len(result) != len(in_degree):
        return ""

    return "".join(result)

Common mistakes:

  • Forgetting the "prefix" edge case: ["abc", "ab"] is invalid because a longer word cannot come before its prefix
  • Comparing ALL character pairs instead of just the first difference between adjacent words
  • Not initializing in_degree for characters that appear in words but have no ordering constraints
Network Delay Time (#743) -- Dijkstra Application ▼

The classic "apply Dijkstra" problem: find the time it takes for a signal to reach all nodes. The answer is the maximum shortest-path distance.

import heapq
from collections import defaultdict

def network_delay_time(times, n, k):
    """
    LeetCode 743: Network Delay Time
    times: [[u, v, w], ...] -- directed weighted edges
    n: number of nodes (1-indexed)
    k: source node

    Return: time for signal from k to reach ALL nodes, or -1.

    This is textbook Dijkstra. The answer is max(all shortest distances).
    """
    graph = defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))

    dist = {}
    heap = [(0, k)]

    while heap:
        d, u = heapq.heappop(heap)

        if u in dist:
            continue  # Already settled

        dist[u] = d

        for v, w in graph[u]:
            if v not in dist:
                heapq.heappush(heap, (d + w, v))

    # All n nodes must be reachable
    if len(dist) != n:
        return -1

    return max(dist.values())

Why this matters: This is the "hello world" of Dijkstra problems. If you can write this cleanly in an interview, you've demonstrated that you understand weighted shortest paths. Interviewers often start here and then add constraints (limited stops = Bellman-Ford, bidirectional = two Dijkstras, etc.).

Interview Follow-up Questions

Common Follow-ups & What Interviewers Look For ▼

1. "How would you detect a cycle in a directed vs undirected graph?"

What they're looking for:

  • Undirected: DFS with parent tracking, or Union-Find
  • Directed: DFS with 3 colors (white/gray/black) or topological sort
  • Why the difference: back edge to parent is not a cycle in undirected graphs

Red flags: Using the same algorithm for both, not understanding edge classification.

2. "When would you use adjacency matrix vs adjacency list?"

What they're looking for:

  • Matrix: Dense graphs, O(1) edge lookup needed, small graphs
  • List: Sparse graphs (most real-world graphs), memory efficiency
  • Real numbers: Social network with 1M users has 1M nodes but avg 500 edges each

Red flags: Not knowing space complexity: O(V^2) vs O(V+E).

3. "How would you handle negative edge weights?"

What they're looking for:

  • Dijkstra fails with negative edges
  • Bellman-Ford: O(VE), handles negative weights, detects negative cycles
  • Real use case: arbitrage detection in currency exchange

Red flags: Not knowing Dijkstra's limitation.

4. "How does Union-Find achieve near O(1) operations?"

What they're looking for:

  • Path compression: flatten tree on find()
  • Union by rank: attach smaller tree under larger
  • Combined: O(alpha(n)) - inverse Ackermann, practically O(1)

Red flags: Not knowing both optimizations are needed together.

5. "How would you find shortest path in a graph with billions of nodes?"

What they're looking for:

  • Precomputation: Contraction Hierarchies, ALT, Hub Labeling
  • Partitioning: Divide graph into regions, precompute boundary distances
  • Trade-off: preprocessing time vs query time

Red flags: Only knowing Dijkstra/BFS without discussing scale.
