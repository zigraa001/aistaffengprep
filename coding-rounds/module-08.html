LRU Cache

Design & Implementation ▼

Requirements: O(1) get and put operations, evict least recently used when at capacity.

class Node:
    def __init__(self, key=0, val=0):
        self.key = key
        self.val = val
        self.prev = None
        self.next = None

class LRUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = {}  # key -> Node

        # Dummy head and tail for easier manipulation
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head

    def _add_to_head(self, node):
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def _remove(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        node = self.cache[key]
        self._remove(node)
        self._add_to_head(node)
        return node.val

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            self._remove(self.cache[key])

        node = Node(key, value)
        self.cache[key] = node
        self._add_to_head(node)

        if len(self.cache) > self.capacity:
            lru = self.tail.prev
            self._remove(lru)
            del self.cache[lru.key]

import time
from collections import deque

class RateLimiter:
    """
    Rate limiter: allow 'limit' requests per 'window' seconds.
    Note: Not thread-safe as shown. In production, add a lock
    per user_id or use atomic operations (e.g., Redis MULTI/EXEC).
    """
    def __init__(self, limit: int, window_seconds: int):
        self.limit = limit
        self.window = window_seconds
        self.requests = {}  # user_id -> deque of timestamps

    def is_allowed(self, user_id: str) -> bool:
        now = time.time()
        window_start = now - self.window

        if user_id not in self.requests:
            self.requests[user_id] = deque()

        # Remove old requests outside window
        while self.requests[user_id] and self.requests[user_id][0] < window_start:
            self.requests[user_id].popleft()

        # Check limit
        if len(self.requests[user_id]) >= self.limit:
            return False

        # Record request
        self.requests[user_id].append(now)
        return True

# Trade-off: This "sliding window log" stores every timestamp.
# Memory is O(limit) per user in steady state after cleanup.
# Alternatives:
# - Token bucket: O(1) space per user, allows short bursts
# - Sliding window counter: O(1) space, approximate but cheaper
# - Fixed window: simplest but allows 2x burst at window boundaries

Log Parser

Thread-Safe Implementation ▼ 
import threading
import re
from collections import defaultdict
from datetime import datetime

class LogParser:
    LOG_PATTERN = re.compile(
        r'^\[(?P[\d\-T:.]+)\]\s+'
        r'(?P\w+)\s+'
        r'(?P\w+):\s+'
        r'(?P.*)'
    )

    def __init__(self):
        self._lock = threading.Lock()
        self._service_counts = defaultdict(int)
        self._error_counts = defaultdict(int)

    def ingest(self, line: str) -> bool:
        match = self.LOG_PATTERN.match(line.strip())
        if not match:
            return False

        with self._lock:
            service = match.group('service')
            level = match.group('level')
            self._service_counts[service] += 1
            if level == 'ERROR':
                self._error_counts[service] += 1

        return True

    def get_error_rate(self, service: str) -> float:
        with self._lock:
            total = self._service_counts.get(service, 0)
            errors = self._error_counts.get(service, 0)
            return errors / total if total > 0 else 0.0

In-Memory File System

Design & Implementation (Meta / Google favorite) ▼

Problem: Implement an in-memory file system with ls, mkdir, addContentToFile, and readContentFromFile. This is a classic Meta and Google interview question that tests your ability to design clean data structures and APIs.

Why companies ask this: It evaluates OOP design, tree data structures in practice, and whether you can handle edge cases like root directory, nested paths, and sorted output. Meta asks variants of this in ~15% of onsite coding rounds.

class TrieNode:
    """Each node represents a directory or file in the filesystem."""
    def __init__(self):
        self.children = {}      # name -> TrieNode
        self.content = ""       # non-empty only for files
        self.is_file = False

class FileSystem:
    def __init__(self):
        self.root = TrieNode()

    def _traverse(self, path: str) -> TrieNode:
        """Navigate to the node at the given path."""
        node = self.root
        if path == "/":
            return node
        parts = path.split("/")[1:]  # skip empty string before leading "/"
        for part in parts:
            if part not in node.children:
                node.children[part] = TrieNode()
            node = node.children[part]
        return node

    def ls(self, path: str) -> list[str]:
        """
        If path is a file, return [filename].
        If path is a directory, return sorted list of entries.
        """
        node = self._traverse(path)
        if node.is_file:
            # Return just the file name (last component)
            return [path.split("/")[-1]]
        return sorted(node.children.keys())

    def mkdir(self, path: str) -> None:
        """Create directory path; intermediate dirs are created automatically."""
        self._traverse(path)  # _traverse creates nodes along the way

    def addContentToFile(self, filePath: str, content: str) -> None:
        """Append content to file. Creates the file if it doesn't exist."""
        node = self._traverse(filePath)
        node.is_file = True
        node.content += content

    def readContentFromFile(self, filePath: str) -> str:
        """Return the content of the file at filePath."""
        node = self._traverse(filePath)
        return node.content


# --- Usage ---
fs = FileSystem()
fs.mkdir("/a/b/c")
fs.addContentToFile("/a/b/c/d.txt", "hello ")
fs.addContentToFile("/a/b/c/d.txt", "world")
print(fs.ls("/a/b/c"))                    # ['d.txt']
print(fs.readContentFromFile("/a/b/c/d.txt"))  # 'hello world'
print(fs.ls("/a/b"))                       # ['c']
print(fs.ls("/"))                          # ['a']

Data structure Trie (tree of nodes) HashMap <path, content> -- O(1) lookup but ls() requires scanning all keys
ls() complexity O(depth + k log k) where k = children count With SortedDict children: O(depth + k) no sorting needed

Companies: Meta (very common), Google, Amazon. Often extended with rm, mv, or wildcard search in follow-ups.

Iterator Design (Flatten Nested List / Zigzag / Peeking)

Multiple Iterator Patterns ▼

Problem family: Implement iterators that lazily traverse complex data structures. These test your understanding of the iterator pattern, stack-based traversal, and lazy evaluation -- concepts that matter in production code (streaming, pagination, data pipelines).

1. Flatten Nested List Iterator (LeetCode #341)

Given a nested list of integers, implement an iterator to flatten it. Each element is either an integer or a list whose elements may also be integers or other lists.

class NestedIterator:
    """
    Approach: Use a stack and lazily flatten. We push elements in
    reverse order so the first element is on top.

    Why lazy? If the nested list has millions of elements, we don't
    want to flatten everything upfront. This is O(1) amortized per
    next() call and O(D) space where D = max nesting depth.
    """
    def __init__(self, nestedList):
        # Stack stores elements in reverse for correct ordering
        self.stack = list(reversed(nestedList))

    def next(self) -> int:
        # hasNext() guarantees top of stack is an integer
        return self.stack.pop().getInteger()

    def hasNext(self) -> bool:
        # Keep flattening until top is an integer or stack is empty
        while self.stack:
            top = self.stack[-1]
            if top.isInteger():
                return True
            # Pop the list and push its elements in reverse
            self.stack.pop()
            for elem in reversed(top.getList()):
                self.stack.append(elem)
        return False


# --- Peeking Iterator (wraps any iterator with peek()) ---
class PeekingIterator:
    """
    Wraps any iterator to support peek() without consuming.
    Used in merge operations, parsers, and stream processing.
    """
    def __init__(self, iterator):
        self.iterator = iterator
        self._has_peeked = False
        self._peeked_value = None

    def peek(self):
        if not self._has_peeked:
            self._peeked_value = self.iterator.next()
            self._has_peeked = True
        return self._peeked_value

    def next(self):
        if self._has_peeked:
            self._has_peeked = False
            val = self._peeked_value
            self._peeked_value = None
            return val
        return self.iterator.next()

    def hasNext(self):
        return self._has_peeked or self.iterator.hasNext()


# --- Zigzag Iterator ---
class ZigzagIterator:
    """
    Alternates between two (or k) lists:
    v1 = [1, 2], v2 = [3, 4, 5, 6] -> 1, 3, 2, 4, 5, 6

    Uses a deque of iterators. Pop from front, yield next value,
    push back if iterator has more elements. Generalizes to k lists.
    """
    def __init__(self, v1, v2):
        from collections import deque
        self.queue = deque()
        if v1:
            self.queue.append(iter(v1))
        if v2:
            self.queue.append(iter(v2))

    def next(self) -> int:
        it = self.queue.popleft()
        val = next(it)
        # Put iterator back if it has more elements
        # We use a sentinel to check without consuming
        try:
            peeked = next(it)
            from itertools import chain
            self.queue.append(chain([peeked], it))
        except StopIteration:
            pass
        return val

    def hasNext(self) -> bool:
        return len(self.queue) > 0

Trade-offs & Discussion

Iterator Time per call Space Key insight
NestedIterator O(1) amortized O(D) depth Lazy flatten with reverse-push on stack
PeekingIterator O(1) O(1) extra Cache one element; decorator pattern
ZigzagIterator O(1) O(k) iterators Round-robin with deque of iterators

Companies: Google (NestedIterator is a top-10 most asked), Meta, Apple, LinkedIn. Zigzag and Peeking variants appear at Amazon and Microsoft.

Time-Based Key-Value Store

LeetCode #981 - Full Implementation with Binary Search ▼

Problem: Design a key-value store where each key can have multiple values at different timestamps. set(key, value, timestamp) stores the value, and get(key, timestamp) returns the value with the largest timestamp ≤ the given timestamp.

Why this matters: This is a simplified model of how versioned data stores work (think Git, database MVCC, or feature flag systems). It tests your ability to combine hash maps with binary search.

import bisect

class TimeMap:
    """
    Time-based key-value store.

    Design decisions:
    - HashMap of key -> list of (timestamp, value) pairs
    - Timestamps are strictly increasing per key (problem guarantee)
      so each list is already sorted -> binary search works directly
    - bisect_right finds insertion point; we want the element just before

    Time: set O(1), get O(log n) where n = values for that key
    Space: O(total set calls)
    """
    def __init__(self):
        self.store = {}  # key -> [(timestamp, value), ...]

    def set(self, key: str, value: str, timestamp: int) -> None:
        if key not in self.store:
            self.store[key] = []
        self.store[key].append((timestamp, value))

    def get(self, key: str, timestamp: int) -> str:
        if key not in self.store:
            return ""

        pairs = self.store[key]
        # Binary search: find rightmost entry with ts <= timestamp
        # bisect_right on timestamps gives us the insertion point
        timestamps = [p[0] for p in pairs]
        idx = bisect.bisect_right(timestamps, timestamp)

        if idx == 0:
            return ""  # all timestamps are after the query
        return pairs[idx - 1][1]


# --- Optimized version avoiding repeated list extraction ---
class TimeMapOptimized:
    """
    Same logic but avoids creating a timestamp list on every get().
    Uses bisect on tuples directly: (timestamp, value) tuples sort
    by timestamp first, which is what we want.
    """
    def __init__(self):
        self.store = {}

    def set(self, key: str, value: str, timestamp: int) -> None:
        if key not in self.store:
            self.store[key] = []
        self.store[key].append((timestamp, value))

    def get(self, key: str, timestamp: int) -> str:
        if key not in self.store:
            return ""

        pairs = self.store[key]
        # Search for (timestamp, chr(127)) -- chr(127) is larger than
        # any reasonable string, so bisect_right finds the right spot
        idx = bisect.bisect_right(pairs, (timestamp, chr(127)))

        if idx == 0:
            return ""
        return pairs[idx - 1][1]


# --- Usage ---
tm = TimeMapOptimized()
tm.set("foo", "bar", 1)
tm.set("foo", "bar2", 4)
print(tm.get("foo", 1))   # "bar"
print(tm.get("foo", 3))   # "bar"  (largest ts <= 3 is 1)
print(tm.get("foo", 4))   # "bar2"
print(tm.get("foo", 5))   # "bar2" (largest ts <= 5 is 4)
print(tm.get("foo", 0))   # ""     (no ts <= 0)

Trade-offs & Follow-ups

Approach set() get() Notes
List + bisect O(1) O(log n) Best for sorted-insert timestamps
SortedList (third-party) O(log n) O(log n) Handles unordered inserts; more memory
TreeMap (Java) O(log n) O(log n) via floorKey() Java-idiomatic; built-in floorEntry

Common follow-ups: What if timestamps are not strictly increasing? (Use SortedList or insort.) What about TTL expiration? (Background cleanup thread or lazy eviction on get.)

Companies: Google, Lyft, Airbnb, Databricks. Often asked as a warm-up for system design discussions about versioned storage.

Design Twitter

LeetCode #355 - OOP + Data Structures Combination ▼

Problem: Design a simplified Twitter where users can post tweets, follow/unfollow other users, and retrieve the 10 most recent tweets in their news feed (from themselves and people they follow).

Why this is a staff-level question: It combines OOP modeling, hash maps, heaps, and linked-list-style thinking. Interviewers evaluate your API design, class decomposition, and ability to optimize the feed generation with a merge-k-sorted-lists approach.

import heapq
from collections import defaultdict

class Twitter:
    """
    Key design decisions:
    1. Global timestamp counter for ordering tweets
    2. Each user's tweets stored as a list (most recent last)
    3. News feed uses a max-heap merge of k sorted lists
       (same pattern as "merge k sorted linked lists")
    4. Users always follow themselves (simplifies feed logic)
    """
    def __init__(self):
        self.timestamp = 0
        self.tweets = defaultdict(list)    # userId -> [(ts, tweetId), ...]
        self.following = defaultdict(set)  # userId -> set of followeeIds

    def postTweet(self, userId: int, tweetId: int) -> None:
        self.tweets[userId].append((self.timestamp, tweetId))
        self.timestamp += 1
        # Ensure user follows themselves
        self.following[userId].add(userId)

    def getNewsFeed(self, userId: int) -> list[int]:
        """
        Merge k sorted lists (each user's tweets in reverse chronological).
        Use a max-heap (negate timestamps for Python's min-heap).

        Time: O(k * log k + 10 * log k) where k = followees count
        Space: O(k) for the heap
        """
        self.following[userId].add(userId)

        # Heap entries: (-timestamp, tweetId, userId, index_in_tweet_list)
        max_heap = []
        for followee in self.following[userId]:
            if self.tweets[followee]:
                idx = len(self.tweets[followee]) - 1
                ts, tid = self.tweets[followee][idx]
                heapq.heappush(max_heap, (-ts, tid, followee, idx))

        result = []
        while max_heap and len(result) < 10:
            neg_ts, tid, uid, idx = heapq.heappop(max_heap)
            result.append(tid)
            # Push the next older tweet from this user
            if idx > 0:
                ts2, tid2 = self.tweets[uid][idx - 1]
                heapq.heappush(max_heap, (-ts2, tid2, uid, idx - 1))

        return result

    def follow(self, followerId: int, followeeId: int) -> None:
        self.following[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int) -> None:
        # Users cannot unfollow themselves
        if followeeId != followerId:
            self.following[followerId].discard(followeeId)


# --- Usage ---
twitter = Twitter()
twitter.postTweet(1, 5)
print(twitter.getNewsFeed(1))        # [5]

twitter.follow(1, 2)
twitter.postTweet(2, 6)
print(twitter.getNewsFeed(1))        # [6, 5] -- most recent first

twitter.unfollow(1, 2)
print(twitter.getNewsFeed(1))        # [5] -- user 2's tweets no longer visible

Trade-offs & Architecture Discussion

Design Choice This Implementation Production Alternative
Feed generation Pull model (merge on read) Push model (fan-out on write) for faster reads
Tweet storage In-memory list per user Sharded DB + Redis timeline cache
Merge algorithm Heap merge of k lists Same pattern! Twitter literally uses this at scale
Celebrity problem Not handled (all users equal) Hybrid: push for normal users, pull for celebrities with millions of followers

Interview tip: After solving the coding problem, mention the pull-vs-push trade-off. This naturally bridges into system design territory and shows architectural awareness -- exactly what staff-level interviews look for.

Companies: Twitter (obviously), Meta, Amazon, LinkedIn. The merge-k-sorted-lists pattern also appears at Google and Apple in different contexts.
