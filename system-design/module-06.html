URL Shortener

Design Overview ▼

Requirements

  • Shorten long URLs to ~7 characters
  • Redirect shortened URLs to original
  • Scale: 500M new URLs/month, 100:1 read/write ratio
flowchart LR C[Client] --> LB[Load Balancer] LB --> API[API Servers] API --> Cache[Redis Cache] API --> DB[(Database)] API --> Counter[ID Generator]

Key Decisions

  • ID Generation: Base62 encoding of counter or UUID
  • Database: NoSQL (simple key-value access)
  • Caching: Cache popular URLs (80/20 rule)

Estimation

Write QPS: 500M / (30 * 24 * 3600) ≈ 200/s
Read QPS: 200 * 100 = 20,000/s
Storage (5 years): 500M * 12 * 5 * 500B ≈ 15TB

Complete URL Shortener Service

# Python - URL Shortener with FastAPI from fastapi import FastAPI, HTTPException from fastapi.responses import RedirectResponse import redis import hashlib from typing import Optional app = FastAPI() redis_client = redis.Redis(host='localhost', port=6379, db=0) ALPHABET = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz" def encode_base62(num: int) -> str: if num == 0: return ALPHABET[0] result = [] while num > 0: result.append(ALPHABET[num % 62]) num //= 62 return ''.join(reversed(result)).zfill(7) @app.post("/api/shorten") async def shorten_url(long_url: str, custom_alias: Optional[str] = None): # Custom alias if custom_alias: if redis_client.exists(f"url:{custom_alias}"): raise HTTPException(400, "Alias already exists") short_code = custom_alias else: # Generate using atomic counter counter = redis_client.incr("url_counter") short_code = encode_base62(counter) # Store mapping redis_client.set(f"url:{short_code}", long_url) return {"short_url": f"https://short.url/{short_code}"} @app.get("/{short_code}") async def redirect(short_code: str): long_url = redis_client.get(f"url:{short_code}") if not long_url: raise HTTPException(404, "URL not found") # Increment click counter (async in production) redis_client.incr(f"clicks:{short_code}") return RedirectResponse(url=long_url.decode(), status_code=302)

Distributed Rate Limiter with Redis

# Python - Sliding Window Counter Rate Limiter import time import redis class SlidingWindowRateLimiter: """ Sliding window counter - best balance of accuracy and memory. Smooths the boundary issue of fixed window. """ def __init__(self, redis_client: redis.Redis, limit: int, window_seconds: int): self.redis = redis_client self.limit = limit self.window = window_seconds def is_allowed(self, key: str) -> bool: now = time.time() current_window = int(now // self.window) prev_window = current_window - 1 position_in_window = (now % self.window) / self.window # Keys for current and previous window curr_key = f"rate:{key}:{current_window}" prev_key = f"rate:{key}:{prev_window}" # Get counts prev_count = int(self.redis.get(prev_key) or 0) curr_count = int(self.redis.get(curr_key) or 0) # Weighted count: prev * (1 - position) + current weighted_count = prev_count * (1 - position_in_window) + curr_count if weighted_count >= self.limit: return False # Increment current window pipe = self.redis.pipeline() pipe.incr(curr_key) pipe.expire(curr_key, self.window * 2) # Keep for 2 windows pipe.execute() return True # Usage limiter = SlidingWindowRateLimiter(redis.Redis(), limit=100, window_seconds=60) if limiter.is_allowed(user_id): process_request() else: return_429_error()

Key Components

  • Message Queue: Decouple sending from delivery
  • Presence Service: Track online/offline status
  • Message Store: Cassandra or similar for chat history

Message Routing Implementation

# Python - Chat message routing with Redis pub/sub import asyncio import json import redis.asyncio as redis from dataclasses import dataclass from typing import Dict, Set @dataclass class ChatMessage: message_id: str sender_id: str recipient_id: str content: str timestamp: int class ChatRouter: def __init__(self, redis_url: str, server_id: str): self.redis = redis.from_url(redis_url) self.server_id = server_id self.local_connections: Dict[str, Set] = {} # user_id -> websockets async def register_connection(self, user_id: str, websocket): """Register a user's WebSocket connection on this server.""" if user_id not in self.local_connections: self.local_connections[user_id] = set() self.local_connections[user_id].add(websocket) # Store user -> server mapping in Redis await self.redis.hset("user:servers", user_id, self.server_id) # Subscribe to messages for this user pubsub = self.redis.pubsub() await pubsub.subscribe(f"user:{user_id}:messages") async def send_message(self, message: ChatMessage): """Route message to recipient.""" # 1. Persist to database (async) await self._persist_message(message) # 2. Find recipient's server recipient_server = await self.redis.hget("user:servers", message.recipient_id) if recipient_server == self.server_id: # Recipient on same server - deliver directly await self._deliver_locally(message) elif recipient_server: # Recipient on different server - use pub/sub await self.redis.publish( f"user:{message.recipient_id}:messages", json.dumps(message.__dict__) ) else: # Recipient offline - queue for later await self._queue_for_offline(message) async def _deliver_locally(self, message: ChatMessage): """Deliver to local WebSocket connections.""" connections = self.local_connections.get(message.recipient_id, set()) for ws in connections: await ws.send_json(message.__dict__) async def _queue_for_offline(self, message: ChatMessage): """Store message for offline delivery.""" await self.redis.lpush( f"user:{message.recipient_id}:pending", json.dumps(message.__dict__) )
